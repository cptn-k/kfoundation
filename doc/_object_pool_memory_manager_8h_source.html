99  void ObjectPoolMemoryManager<T>::grow() {
100  int newSize = _size * _growthRate;
101  ObjectRecord* newTable = new ObjectRecord[newSize];
102  memcpy((void*)newTable, (void*)_table, sizeof(ObjectRecord) * _size);
103 
104  delete[] _table;
105  _table = newTable;
106 
107  LOG << "ObjectPool " << _id << " resized from " << _size
108  << " to " << newSize << ENDS;
109 
110  for(int i = _size; i < newSize; i++) {
111  ObjectRecord& record = newTable[i];
112  record.retainCount = -1;

114  record.index = i;
115  record.serialNumber = -1;
116  record.isStatic = false;
117  record.ptr = new T(_id, i);
118  }
119 
120  _size = newSize;
121  _master->updataTable(_id);
122  }

143  Ref<T> ObjectPoolMemoryManager<T>::get() {
144  pthread_mutex_lock(&_mutex);
145  int index = _next;
146  ObjectRecord& rec = _table[index];
147  rec.retainCount = 1;
148  rec.serialNumber = _serialCounter++;
149  rec.key = rand() % 65530 - 32760;
150  rec.manager = _id;
151  rec.isStatic = false;
152 
153  _count++;
154 
155  if(_count == _size) {
156  grow();
157  }

160  _next++;

165  pthread_mutex_unlock(&_mutex);

209  void ObjectPoolMemoryManager<T>::release(kf_int32_t index, kf_int16_t key) {
210 #ifdef DEBUG
211  string recStr;
212  if(_trace) {
213  recStr = toString(index);
214  }
215 #endif
216 
217  // synchronized {
218  pthread_mutex_lock(&_mutex);
219  ObjectRecord& record = _table[index];
220  if(record.key != key) {
221  pthread_mutex_unlock(&_mutex);
222  throw InvalidPointerException("The pointer being released is invalid: "
223  + Int(_id) + ":" + Int(index));
224  }
225  if(!record.isStatic) {
226  record.retainCount--;
227  if(record.retainCount == 0) {
228  record.retainCount = -1;
229  record.key = rand() % 65530 - 32760;
230  _count--;
231  } else if(record.retainCount < 0) {
232  pthread_mutex_unlock(&_mutex);
233  throw InvalidPointerException("Object is released too many times: "
234  + Int(_id) + ":" + Int(index));
235  }
236  }
237  pthread_mutex_unlock(&_mutex);
238  // } synchronized
239 
240  if(record.retainCount == -1) {
241  dynamic_cast<PoolObject*>(record.ptr)->finalize();
242  }
243 
244 #ifdef DEBUG
245  if(_trace) {
246  string x;
247  if(record.retainCount == -1) {
248  x = " X ";
249  }
250  LOG << "Released: " << recStr << " --> " << record.retainCount << x << ENDS;
251  }
252 #endif
253  }
