359  template<typename T>
360  void RefArrayBase<T>::serialize(Ref<ObjectSerializer> builder) const {
361  builder->collection();
362  size_t s = _size;
363  for(int i = 0; i < s; i++) {
364  T d = _data[i];
365  if(d.isNull()) {
366  builder->null();
367  } else if(d.template ISA(SerializingStreamer)) {
368  builder->object(dynamic_cast<const SerializingStreamer&>(*d));
369  } else if(d.template ISA(Streamer)) {
370  const Streamer& streamer = dynamic_cast<const Streamer&>(*d);
371  builder->object(d.getTypeName())
372  ->attribute(K"value", streamer.toString())
373  ->endObject();
374  } else {
375  builder->object(d.getTypeName())->endObject();
376  }
377  }
378  builder->endCollection();
379  }
