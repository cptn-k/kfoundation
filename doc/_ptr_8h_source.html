367  int Ptr<T>::getRetainCount() const {
368  if(!_locator.objectIndex) {

372  ObjectRecord* record = *(objectTable + _locator.managerIndex)
373  + _locator.objectIndex;

376  return RETAIN_COUNT_INVALID;
