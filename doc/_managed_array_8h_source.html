92  template<typename T>
93  void ManagedArray<T>::grow(kf_int32_t newCapacity) {
94  _capacity = newCapacity;
95  Ptr<T>* newData = new Ptr<T>[_capacity];
96  for(int i = 0; i < _size; i++) {
97  newData[i] = _data[i];
98  }
99  LOG << "ManagedArray resized to " << _capacity << EL;
100  delete[] _data;
101  _data = newData;
102  }

297  template<typename T>
298  void ManagedArray<T>::serialize(PPtr<ObjectSerializer> builder) const {
299  builder->collection();
300  size_t s = _size;
301  for(int i = 0; i < s; i++) {
302  ManagedObject* obj = _data[i].toPurePtr();
303  if(INSTANCEOF(obj, SerializingStreamer)) {
304  dynamic_cast<SerializingStreamer*>(obj)->serialize(builder);
305  } else if(INSTANCEOF(obj, Streamer)) {
306  builder->object("Item")
307  ->attribute("value", dynamic_cast<Streamer*>(obj)->toString())
308  ->endObject();
309  } else if(obj == NULL) {
310  builder->null()->endObject();
311  } else {
312  builder->object(System::demangle(typeid(T).name()))->endObject();
313  }
314  }
315  builder->endCollection();
316  }
